    this._input = value.toLowerCase();
  }

    if (typeof this._input === "string") {
      return this._input.toUpperCase();
    } else {
      return "value should be 'string'";
    }
  }

  set uppercaseValue(value: string) {
    this._input = value.toUpperCase();
  }
}
